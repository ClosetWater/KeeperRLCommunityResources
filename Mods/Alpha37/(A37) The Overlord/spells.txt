Def TryTrowFlame()
  TryNoFriendlyFire(
    {
      CreatureMessage {"were hit by the flaming projectile" "was hit by the flaming projectile"}
      Damage FIRE_DAMAGE SHOOT
      Fire{}
    },
    CreatureMessage {"were missed by the flaming projectile" "was missed by the flaming projectile"}
  )
End
