#Define what a standard bonus mod biome looks like:
#Ensure the basic biomes have the z-level enemies, wildlife and some basic enemies sorted out.
###################

Def STD_ENEMIES()
    { id = "CORPSES" probability = 0.2 }
    { id = "RUINS" probability = 0.2 }
	{ id = "BONUS_SHADOW_TEMPLE" probability = 0.2 }
    { id = "BONUS_WILDMEN_CAVE_HIDDEN" probability = 0.2 }
	{ id = "BOTTOMLESS_DUNGEON"  probability = 0.2 }
End

Def KEEPER_ENEMIES()
    { id = "BONUS_WILDMEN_CAVE_HIDDEN" }
	{ id = "BOTTOMLESS_DUNGEON" }
End

Def STD_WILDLIFE()
      1 "BOAR"
	  1 "DEER"
	  1 "RAVEN"
	  1 "SPIDER"
	  1 "SNAKE"
	  1 "FOX"
	  1 "RAT"
	  1 "DONKEY"
End

Def STANDARD_BIOME(the_name, the_viewId, the_water, the_sand, the_items, creatures, enemy1, enemy2, enemy3, enemy4, boss)

  #Basics

  name = the_name
  viewId = { the_viewId }
  overrideWaterType = the_water
  sandType = the_sand
  items = the_items

  #Add the standard mountain enemies but add a boss that is specified.
  mountainEnemies = {
    {1 8} {id = enemy1 probability = 0.2 }
    {1 8} {id = enemy2 probability = 0.2 }
    {1 8} {id = enemy3 probability = 0.2 }
	{1 8} {id = enemy4 probability = 0.2 }
    10 {id = boss}
  }
End

Def PAINT(the_density, the_painted, comment, the_background)
    { ratio = 0.1 density = the_density onType = the_background trees = the_painted }
End
